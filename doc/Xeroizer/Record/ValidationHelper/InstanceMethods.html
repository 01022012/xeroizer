def errors_for(attribute)
  if errors.is_a?(Array)
    errors.find_all { | (attr, msg) | attr == attribute }.map { | (attr, msg) | msg }
  end
end

def valid?
  @errors = []
  if self.class.validators
    self.class.validators.each do | validator |
      validator.validate(self)
    end
  end
  @errors.size == 0
end

def to_xml(b = Builder::XmlMarkup.new(:indent => 2))
  b.tag!(parent.model_name) { 
    attributes.each do | key, value |
      unless value.nil?
        field = self.class.fields[key]
        xml_value_from_field(b, field, value)
      end
    end
  }
end
